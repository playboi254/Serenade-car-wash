/* ─── Drag-scroll carousel ─── */
function dragCarousel(wrapId, trackId) {
  const wrap  = document.getElementById(wrapId);
  const track = document.getElementById(trackId);
  if (!wrap || !track) return;

  let down = false, startX = 0, baseX = 0;

  const clamp = (v) => {
    const max = track.scrollWidth - wrap.clientWidth;
    return Math.max(0, Math.min(v, max));
  };
  const applyX = (x) => { track.style.transform = `translateX(-${x}px)`; wrap._x = x; };

  wrap.addEventListener('mousedown',  e => { down = true; startX = e.pageX; baseX = wrap._x || 0; wrap.style.cursor='grabbing'; e.preventDefault(); });
  document.addEventListener('mouseup', () => { down = false; wrap.style.cursor='grab'; });
  document.addEventListener('mousemove', e => {
    if (!down) return;
    applyX(clamp(baseX - (e.pageX - startX) * 1.3));
  });

  let tStartX = 0, tBaseX = 0;
  wrap.addEventListener('touchstart',  e => { tStartX = e.touches[0].pageX; tBaseX = wrap._x || 0; }, {passive:true});
  wrap.addEventListener('touchmove',   e => { applyX(clamp(tBaseX - (e.touches[0].pageX - tStartX))); }, {passive:true});
}
